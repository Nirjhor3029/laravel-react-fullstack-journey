Topic 2: Creating Your First Component
A React component is a reusable piece of UI.

// Functional Component
function Welcome() {
    return <h1>Hello, Welcome!</h1>;
}

// Arrow Function Component
const Welcome = () => {
    return <h1>Hello, Welcome!</h1>;
};

// Using the component
function App() {
    return (
        <div>
            <Welcome />
            <Welcome />
            <Welcome />
        </div>
    );
}
